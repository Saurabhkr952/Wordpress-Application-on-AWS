1.
# create the html directory and mount the efs to it
sudo su
yum update -y
mkdir -p /var/www/html
sudo mount -t nfs4 -o nfsvers=4.1,rsize=1048576,wsize=1048576,hard,timeo=600,retrans=2,noresvport fs-aed8ad5b.efs.us-east-1.amazonaws.com:/ /var/www/html


2.
# install apache 
sudo yum install -y httpd httpd-tools mod_ssl
sudo systemctl enable httpd 
sudo systemctl start httpd

3
# install php 7.4
sudo amazon-linux-extras enable php7.4
sudo yum clean metadata
sudo yum install php php-common php-pear -y
sudo yum install php-{cgi,curl,mbstring,gd,mysqlnd,gettext,json,xml,fpm,intl,zip} -y

4
# install mysql5.7
sudo rpm -Uvh https://dev.mysql.com/get/mysql57-community-release-el7-11.noarch.rpm

sudo yum install mysql-community-server -y
sudo systemctl enable mysqld
sudo systemctl start mysqld

5
# set permissions
sudo usermod -a -G apache ec2-user
sudo chown -R ec2-user:apache /var/www                 # -R recursive                        
sudo chmod 2775 /var/www && find /var/www -type d -exec sudo chmod 2775 {} \;
sudo find /var/www -type f -exec sudo chmod 0664 {} \;
sudo chown apache:apache -R /var/www/html 

6.
# download wordpress files
sudo wget https://wordpress.org/latest.tar.gz                                            # sudo added
sudo tar -xzf latest.tar.gz
sudo cp -r wordpress/* /var/www/html/


7.
# create the wp-config.php file
sudo cp /var/www/html/wp-config-sample.php /var/www/html/wp-config.php


8.
# edit the wp-config.php file
sudo nano /var/www/html/wp-config.php

sed 's/database_name_here/demodb/' /var/www/html/wp-config.php 
sed 's/username_here/user/' /var/www/html/wp-config.php 
sed 's/password_here/QxMCi6rO7T10HybO/' /var/www/html/wp-config.php


9
# restart the webserver
sudo service httpd restart
